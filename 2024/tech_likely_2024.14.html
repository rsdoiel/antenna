use File::Basename;

if (@ARGV == 0) {
  die "Usage: to-pdf file.txt ...\n";

  my ($name, $path, $suffix) = fileparse($_, ".txt");

<pre style="font-family: Iosevka; font-size: 10pt">
BEG

  for (<$text>) { print $html $_ }; # copy text

  print $html <<'END';
